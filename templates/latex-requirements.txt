# LaTeX package reference (informational).
#
# Canonical engine is Tectonic (XeTeX): it AUTO-DOWNLOADS every package below on
# first compile, so there is nothing to install manually. This list just documents
# what the templates use. Under the pdfLaTeX fallback (\ifPDFTeX branch) you need a
# TeX distribution (TeX Live / MiKTeX) that provides these.

# --- Engine & fonts (dual-engine) ---
iftex             # engine detection (\ifPDFTeX guard)
fontspec          # XeTeX/LuaTeX font loading (Tectonic path)
tex-gyre          # TeX Gyre Termes OTF — Times-compatible, REAL bold for ž/ć/š/đ
newtxtext         # pdfLaTeX fallback font (only inside \ifPDFTeX)

# --- Core ---
babel             # croatian language + hyphenation (babel-croatian)
geometry
graphicx
setspace          # 1.5 spacing (thesis/seminar), single (paper)
titlesec          # section heading formatting
caption           # figure/table caption formatting
fancyhdr          # headers / footers

# --- Tables ---
booktabs
longtable
array
tabularx

# --- Math ---
amsmath
amssymb

# --- Counters / misc ---
chngcntr          # chapter-based figure/table/equation numbering
xcolor            # visible warnings for unfilled placeholders
hyperref          # clickable links
abstract          # fsb-paper abstract formatting

# --- Demo/graphs (optional) ---
tikz
pgfplots
